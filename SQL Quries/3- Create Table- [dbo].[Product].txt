CREATE TABLE Product
(
	Id INT IDENTITY(1,1) PRIMARY KEY,
	[Name] NVARCHAR(500) NOT NULL,
	[Description] NVARCHAR(MAX) NULL,
	CategoryId INT NOT NULL DEFAULT 0,				-- DROPDOWNs
	SupplierId INT NOT NULL DEFAULT 0,				-- DROPDOWNs
	SalePrice DECIMAL NOT NULL DEFAULT 0,			
	CostPrice DECIMAL NOT NULL DEFAULT 0,
	
	-- back end fields
	CreatedBy INT NOT NULL DEFAULT 0,
	CreatedOnUtc DATETIME NOT NULL,
	UpdatedBy INT NOT NULL DEFAULT 0,
	UpdatedOnUtc DATETIME NULL,
	Deleted BIT NOT NULL DEFAULT 0,
	-- back end fields

	OutOfStock BIT NOT NULL DEFAULT 0
)
